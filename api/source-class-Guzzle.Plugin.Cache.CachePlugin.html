 59:      * - CacheStorageInterface     storage:       (optional) Adapter used to cache responses

 63:      *                                                       must set to 0 or it will assume the default of 3600 secs.
 64:      * - bool                      debug_headers: (optional) Add debug headers to the response (default true)
 65:      *
 66:      * @param array|CacheAdapterInterface|CacheStorageInterface $options Array of options for the cache plugin,

 68:      *

 70:      */

 72:     {
 73:         if (!is_array($options)) {
 74:             if ($options instanceof CacheAdapterInterface) {
 75:                 $options = array('adapter' => $options);
 76:             } elseif ($options instanceof CacheStorageInterface) {
 77:                 $options = array('storage' => $options);
 78:             } elseif (class_exists('Doctrine\Common\Cache\ArrayCache')) {
 79:                 $options = array('storage' => new DefaultCacheStorage(new DoctrineCacheAdapter(new ArrayCache()), 3600));
 80:             } else {
 81:                 // @codeCoverageIgnoreStart
 82:                 throw new InvalidArgumentException('No cache was provided and Doctrine is not installed');
 83:                 // @codeCoverageIgnoreEnd
 84:             }
 85:         }
 86: 
 87:         // Add a cache storage if a cache adapter was provided
 88:         if (!isset($options['adapter'])) {
 89:             $this->storage = $options['storage'];
 90:         } else {
 91:             $this->storage = new DefaultCacheStorage(
 92:                 $options['adapter'],
 93:                 array_key_exists('default_ttl', $options) ? $options['default_ttl'] : 3600
 94:             );
 95:         }
 96: 
 97:         // Use the provided key provider or the default
 98:         if (!isset($options['key_provider'])) {
 99:             $this->keyProvider = new DefaultCacheKeyProvider();

155: 
156:         // Intercept PURGE requests
157:         if ($request->getMethod() == 'PURGE') {
158:             $this->purge($request);
159:             $request->setResponse(new Response(200, array(), 'purged'));

161:         }

163:         if (!$this->canCache->canCacheRequest($request)) {
164:             return;

166:

276:     /**
277:      * Check if a cache response satisfies a request's caching constraints
278:      *
279:      * @param RequestInterface $request  Request to validate
280:      * @param Response         $response Response to validate
281:      *
282:      * @return bool
283:      */
284:     public function canResponseSatisfyRequest(RequestInterface $request, Response $response)
285:     {
286:         $responseAge = $response->getAge();
287: 
288:         // Check the request's max-age header against the age of the response
289:         if ($request->hasCacheControlDirective('max-age') &&
290:             $responseAge > $request->getCacheControlDirective('max-age')) {
291:             return false;
292:         }
293: 
294:         // Check the response's max-age header
295:         if ($response->isFresh() === false) {
296:             $maxStale = $request->getCacheControlDirective('max-stale');
297:             if (null !== $maxStale) {
298:                 if ($maxStale !== true && $response->getFreshness() < (-1 * $maxStale)) {
299:                     return false;
300:                 }
301:             } elseif ($response->hasCacheControlDirective('max-age')
302:                 && $responseAge > $response->getCacheControlDirective('max-age')
303:             ) {
304:                 return false;
305:             }
306:         }
307: 
308:         // Only revalidate GET requests
309:         if ($request->getMethod() == RequestInterface::GET) {
310:             // Check if the response must be validated against the origin server
311:             if ($request->getHeader('Pragma') == 'no-cache' ||
312:                 $request->hasCacheControlDirective('no-cache') ||
313:                 $request->hasCacheControlDirective('must-revalidate') ||
314:                 $response->hasCacheControlDirective('must-revalidate') ||
315:                 $response->hasCacheControlDirective('no-cache')) {
316:                 // no-cache: When no parameters are present, always revalidate
317:                 // When parameters are present in no-cache and the request includes those same parameters, then the
318:                 // response must re-validate. I'll need an example of what fields look like in order to implement a
319:                 // smarter version of no-cache
320: 
321:                 // Requests can decline to revalidate against the origin server by setting the cache.revalidate param:
322:                 // - never - To never revalidate and always contact the origin server
323:                 // - skip  - To skip revalidation and just use what is in cache
324:                 switch ($request->getParams()->get('cache.revalidate')) {
325:                     case 'never':
326:                         return false;
327:                     case 'skip':
328:                         return true;
329:                     default:
330:                         return $this->revalidation->revalidate($request, $response);
331:                 }
332:             }
333:         }
334: 
335:         return true;
336:     }

339:      * Check if a cache response satisfies a failed request's caching constraints
340:      *
341:      * @param RequestInterface $request  Request to validate
342:      * @param Response         $response Response to validate
343:      *
344:      * @return bool
345:      */
346:     public function canResponseSatisfyFailedRequest(RequestInterface $request, Response $response)
347:     {
348:         $requestStaleIfError = $request->getCacheControlDirective('stale-if-error');
349:         $responseStaleIfError = $response->getCacheControlDirective('stale-if-error');
350: 
351:         if (!$requestStaleIfError && !$responseStaleIfError) {
352:             return false;
353:         }
354: 
355:         if ($requestStaleIfError !== true &&
356:             $requestStaleIfError !== null &&
357:             $response->getAge() - $response->getMaxAge() > $requestStaleIfError
358:         ) {
359:             return false;
360:         }
361: 
362:         if ($responseStaleIfError !== true &&
363:             $responseStaleIfError !== null &&
364:             $response->getAge() - $response->getMaxAge() > $responseStaleIfError
365:         ) {
366:             return false;
367:         }
368: 
369:         return true;
370:     }

373:      * Purge a request from the cache storage
374:      *
375:      * @param RequestInterface $request Request to purge
376:      */
377:     public function purge(RequestInterface $request)
378:     {
379:         // If the request has a cache.purge_methods param, then use that, otherwise use the default known methods
380:         $methods = $request->getParams()->get('cache.purge_methods') ?: array('GET', 'HEAD', 'POST', 'PUT', 'DELETE');
381:         foreach ($methods as $method) {
382:             // Clone the request with each method and clear from the cache
383:             $cloned = RequestFactory::getInstance()->cloneRequestWithMethod($request, $method);
384:             $key = $this->keyProvider->getCacheKey($cloned);
385:             $this->storage->delete($key);
386:         }
