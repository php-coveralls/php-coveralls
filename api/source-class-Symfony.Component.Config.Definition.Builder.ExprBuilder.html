 23:     protected $node;
 24:     public $ifPart;
 25:     public $thenPart;

 28:      * Constructor
 29:      *
 30:      * @param NodeDefinition $node The related node

 40:      * @param \Closure $then
 41:      *
 42:      * @return ExprBuilder
 43:      */

 45:     {
 46:         $this->ifPart = function($v) { return true; };
 47: 
 48:         if (null !== $then) {
 49:             $this->thenPart = $then;

 51:

 57:      *

 62:      * @return ExprBuilder
 63:      */

 66:         if (null === $closure) {

 68:         }

 72:         return $this;

224:     {
