 43:      */

 45:     {
 46:         try {
 47:             $revalidate = $this->createRevalidationRequest($request, $response);
 48:             $validateResponse = $revalidate->send();
 49:             if ($validateResponse->getStatusCode() == 200) {
 50:                 return $this->handle200Response($request, $validateResponse);
 51:             } elseif ($validateResponse->getStatusCode() == 304) {
 52:                 return $this->handle304Response($request, $validateResponse, $response);
 53:             }
 54:         } catch (BadResponseException $e) {
 55:             $this->handleBadResponse($e);
 56:         }
 57: 
 58:         // Other exceptions encountered in the revalidation request are ignored
 59:         // in hopes that sending a request to the origin server will fix it
 60:         return false;
 61:     }

 64:      * Handles a bad response when attempting to revalidate
 65:      *
 66:      * @param BadResponseException $e Exception encountered
 67:      *
 68:      * @throws BadResponseException
 69:      */

 71:     {
 72:         // 404 errors mean the resource no longer exists, so remove from
 73:         // cache, and prevent an additional request by throwing the exception
 74:         if ($e->getResponse()->getStatusCode() == 404) {
 75:             $this->storage->delete($this->cacheKey->getCacheKey($e->getRequest()));

 77:         }
 78:     }

 81:      * Creates a request to use for revalidation
 82:      *
 83:      * @param RequestInterface $request  Request
 84:      * @param Response         $response Response to revalidate
 85:      *

 87:      */

 89:     {
 90:         $revalidate = clone $request;
 91:         $revalidate->removeHeader('Pragma')
 92:             ->removeHeader('Cache-Control')
 93:             ->setHeader('If-Modified-Since', $response->getLastModified() ?: $response->getDate());
 94: 
 95:         if ($response->getEtag()) {
 96:             $revalidate->setHeader('If-None-Match', '"' . $response->getEtag() . '"');
 97:         }
 98: 
 99:         // Remove any cache plugins that might be on the request
100:         $revalidate->getEventDispatcher()->removeSubscriber($this->plugin);
101: 
102:         return $revalidate;
103:     }

131:      * @param RequestInterface $request          Request that was sent

133:      * @param Response         $response         Original cached response
134:      *
135:      * @return bool Returns true if valid, false if invalid
136:      */
137:     protected function handle304Response(RequestInterface $request, Response $validateResponse, Response $response)
138:     {
139:         static $replaceHeaders = array('Date', 'Expires', 'Cache-Control', 'ETag', 'Last-Modified');
140: 
141:         // Make sure that this response has the same ETag
142:         if ($validateResponse->getEtag() != $response->getEtag()) {
143:             return false;
144:         }
145:         // Replace cached headers with any of these headers from the
146:         // origin server that might be more up to date
147:         $modified = false;
148:         foreach ($replaceHeaders as $name) {
149:             if ($validateResponse->hasHeader($name)) {
150:                 $modified = true;
151:                 $response->setHeader($name, $validateResponse->getHeader($name));
152:             }
153:         }
154:         // Store the updated response in cache
155:         if ($modified && $response->canCache()) {
156:             $this->storage->cache(
157:                 $this->cacheKey->getCacheKey($request),
158:                 $response,
159:                 $request->getParams()->get('cache.override_ttl')
160:             );
161:         }
162: 
163:         return true;
