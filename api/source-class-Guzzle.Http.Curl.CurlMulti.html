229:     /**
230:      * Get the data from the multi handle
231:      */
232:     protected function perform()
233:     {
234:         if (!$this->requests) {
235:             return;
236:         }
237: 
238:         // Initialize the handles with a very quick select timeout
239:         $active = $mrc = null;
240:         $this->executeHandles($active, $mrc, 0.001);
241:         $event = array('curl_multi' => $this);
242: 
243:         while (1) {
244: 
245:             $this->processMessages();
246: 
247:             // Exit the function if there are no more requests to send
248:             if (!$this->requests) {
249:                 break;
250:             }
251: 
252:             // Notify each request as polling
253:             $blocking = $total = 0;
254:             foreach ($this->requests as $request) {
255:                 $event['request'] = $request;
256:                 $request->dispatch(self::POLLING_REQUEST, $event);
257:                 ++$total;
258:                 // The blocking variable just has to be non-falsey to block the loop
259:                 if ($request->getParams()->hasKey(self::BLOCKING)) {
260:                     ++$blocking;
261:                 }
262:             }
263: 
264:             if ($blocking == $total) {
265:                 // Sleep to prevent eating CPU because no requests are actually pending a select call
266:                 usleep(500);
267:             } else {
268:                 do {
269:                     $this->executeHandles($active, $mrc, 1);
270:                 } while ($active);
271:             }
272:         }
273:     }

280:         // Get messages from curl handles
281:         while ($done = curl_multi_info_read($this->multiHandle)) {

296:      * Execute and select curl handles until there is activity
297:      *
298:      * @param int $active  Active value to update
299:      * @param int $mrc     Multi result value to update
300:      * @param int $timeout Select timeout in seconds
301:      */
302:     private function executeHandles(&$active, &$mrc, $timeout = 1)
303:     {
304:         do {
305:             $mrc = curl_multi_exec($this->multiHandle, $active);
306:         } while ($mrc == CURLM_CALL_MULTI_PERFORM && $active);
307:         $this->checkCurlResult($mrc);
308: 
309:         // @codeCoverageIgnoreStart
310:         // Select the curl handles until there is any activity on any of the open file descriptors
311:         // See https://github.com/php/php-src/blob/master/ext/curl/multi.c#L170
312:         if ($active && $mrc == CURLM_OK && curl_multi_select($this->multiHandle, $timeout) == -1) {
313:             // Perform a usleep if a previously executed select returned -1
314:             // @see https://bugs.php.net/bug.php?id=61141

338:     /**
339:      * Check for errors and fix headers of a request based on a curl response
340:      *
341:      * @param RequestInterface $request Request to process
342:      * @param CurlHandle       $handle  Curl handle object
343:      * @param array            $curl    Array returned from curl_multi_info_read
344:      *
345:      * @throws CurlException on Curl error
346:      */
347:     protected function processResponse(RequestInterface $request, CurlHandle $handle, array $curl)
348:     {
349:         // Set the transfer stats on the response
350:         $handle->updateRequestFromTransfer($request);
351:         // Check if a cURL exception occurred, and if so, notify things
352:         $curlException = $this->isCurlException($request, $handle, $curl);
353: 
354:         // Always remove completed curl handles.  They can be added back again
355:         // via events if needed (e.g. ExponentialBackoffPlugin)
356:         $this->removeHandle($request);
357: 
358:         if (!$curlException) {
359:             $request->setState(RequestInterface::STATE_COMPLETE, array('handle' => $handle));
360:             // Only remove the request if it wasn't resent as a result of the state change
361:             if ($request->getState() != RequestInterface::STATE_TRANSFER) {
362:                 $this->remove($request);
363:             }
364:         } else {
365:             // Set the state of the request to an error
366:             $request->setState(RequestInterface::STATE_ERROR);
367:             // Notify things that listen to the request of the failure
368:             $request->dispatch('request.exception', array(
369:                 'request'   => $this,
370:                 'exception' => $curlException
371:             ));
372: 
373:             // Allow things to ignore the error if possible
374:             $state = $request->getState();
375:             if ($state != RequestInterface::STATE_TRANSFER) {
376:                 $this->remove($request);
377:             }
378:             // The error was not handled, so fail
379:             if ($state == RequestInterface::STATE_ERROR) {
380:                 /** @var CurlException $curlException */
381:                 throw $curlException;
382:             }
383:         }
384:     }

403:      * Check if a cURL transfer resulted in what should be an exception
404:      *
405:      * @param RequestInterface $request Request to check
406:      * @param CurlHandle       $handle  Curl handle object
407:      * @param array            $curl    Array returned from curl_multi_info_read
408:      *

410:      */
411:     private function isCurlException(RequestInterface $request, CurlHandle $handle, array $curl)
412:     {
413:         if (CURLM_OK == $curl['result'] || CURLM_CALL_MULTI_PERFORM == $curl['result']) {
414:             return false;
415:         }
416: 
417:         $handle->setErrorNo($curl['result']);

422:             ->setCurlInfo($handle->getInfo())
423:             ->setError($handle->getError(), $handle->getErrorNo());

426:     }
